// Search.js

import React, { useState, useEffect } from "react";
import axios from 'axios';
import SearchInput from '../search/SearchInput';
import FilterSelect from '../search/FilterSelect';
import FilteredList from '../search/FilteredList';

const API_BASE_URL = process.env.REACT_APP_API_BASE_URL || 'http://localhost:3001';

/**
 * Search component allows users to search and filter wellplates.
 *
 * @component
 * @returns {JSX.Element} The Search component.
 */
const Search = () => {
  /**
   * State for the data fetched from the server.
   * @type {Array}
   * @memberof Search
   */
  const [data, setData] = useState([]);

  /**
   * State for loading status.
   * @type {boolean}
   * @memberof Search
   */
  const [loading, setLoading] = useState(true);

  /**
   * State for error handling.
   * @type {Error|null}
   * @memberof Search
   */
  const [error, setError] = useState(null);

  /**
   * State for search query.
   * @type {string}
   * @memberof Search
   */
  const [query, setQuery] = useState("");

  /**
   * State for selected material filter.
   * @type {string}
   * @memberof Search
   */
  const [selectedMaterial, setSelectedMaterial] = useState("");

  /**
   * State for selected brand filter.
   * @type {string}
   * @memberof Search
   */
  const [selectedBrand, setSelectedBrand] = useState("");

  /**
   * State for selected number of wells filter.
   * @type {string}
   * @memberof Search
   */
  const [selectedNumberOfWells, setSelectedNumberOfWells] = useState("");

  /**
   * Fetch data from the server when the component mounts.
   * @memberof Search
   * @async
   * @function
   * @returns {Promise<void>}
   */
  useEffect(() => {
    const fetchData = async () => {
      try {
        const response = await axios.get(`${API_BASE_URL}/api/wellplates`);
        setData(response.data);
        setLoading(false);
      } catch (error) {
        setError(error);
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  /**
   * Handle change in search query input.
   * @memberof Search
   * @param {Object} e - Event object.
   */
  const handleChange = (e) => {
    setQuery(e.target.value);
  };

  /**
   * Handle change in material filter.
   * @memberof Search
   * @param {Object} e - Event object.
   */
  const handleMaterialChange = (e) => {
    setSelectedMaterial(e.target.value);
  };

  /**
   * Handle change in brand filter.
   * @memberof Search
   * @param {Object} e - Event object.
   */
  const handleBrandChange = (e) => {
    setSelectedBrand(e.target.value);
  };

  /**
   * Handle change in number of wells filter.
   * @memberof Search
   * @param {Object} e - Event object.
   */
  const handleNumberOfWellsChange = (e) => {
    setSelectedNumberOfWells(e.target.value);
  };

  /**
   * Filter the data based on the search query and selected filters.
   * @memberof Search
   * @type {Array}
   */
  const filteredData = data.filter((item) => {
    return (
      item.name.toLowerCase().includes(query.toLowerCase()) &&
      (selectedMaterial ? item.material === selectedMaterial : true) &&
      (selectedBrand ? item.brand === selectedBrand : true) &&
      (selectedNumberOfWells ? item.number_of_wells === parseInt(selectedNumberOfWells) : true)
    );
  });

  /**
   * Get available materials for the filter based on the selected brand and number of wells.
   * @memberof Search
   * @type {Array}
   */
  const availableMaterials = [...new Set(data.filter(item => 
    (!selectedBrand || item.brand === selectedBrand) && 
    (!selectedNumberOfWells || item.number_of_wells === parseInt(selectedNumberOfWells))
  ).map(item => item.material))];

  /**
   * Get available brands for the filter based on the selected material and number of wells.
   * @memberof Search
   * @type {Array}
   */
  const availableBrands = [...new Set(data.filter(item => 
    (!selectedMaterial || item.material === selectedMaterial) && 
    (!selectedNumberOfWells || item.number_of_wells === parseInt(selectedNumberOfWells))
  ).map(item => item.brand))];

  /**
   * Get available numbers of wells for the filter based on the selected material and brand.
   * @memberof Search
   * @type {Array}
   */
  const availableNumberOfWells = [...new Set(data.filter(item => 
    (!selectedMaterial || item.material === selectedMaterial) && 
    (!selectedBrand || item.brand === selectedBrand)
  ).map(item => item.number_of_wells))];

  if (loading) {
    return <div>Loading...</div>;
  }

  if (error) {
    return <div>Error: {error.message}</div>;
  }

  return (
    <div>
      <h1>Search App</h1>
      <SearchInput query={query} onChange={handleChange} />
      <FilterSelect value={selectedMaterial} onChange={handleMaterialChange} options={availableMaterials} defaultOption="All Materials" />
      <FilterSelect value={selectedBrand} onChange={handleBrandChange} options={availableBrands} defaultOption="All Brands" />
      <FilterSelect value={selectedNumberOfWells} onChange={handleNumberOfWellsChange} options={availableNumberOfWells} defaultOption="All Numbers of Wells" />
      <FilteredList filteredData={filteredData} />
    </div>
  );
};

export default Search;
