Module: utils/fullscreen

Utility functions to manage fullscreen mode in browsers.

Source:
  • client/src/utils/fullscreen.js, line 1
Example
import { addFullscreenChangeListener, requestFullscreen, exitFullscreen } from './fullscreen';

// Add fullscreen change listener
const cleanup = addFullscreenChangeListener((isFullscreen) => {
  console.log('Fullscreen mode:', isFullscreen);
});

// Request fullscreen
requestFullscreen(document.getElementById('myElement'));

// Exit fullscreen
exitFullscreen();

// Cleanup event listener
cleanup();
